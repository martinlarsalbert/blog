fig,ax=plt.subplots()
ax.plot(xs0,ys0,'o-');

dxs0 = np.diff(xs0)
dys0 = np.diff(ys0)
ls0 = np.sqrt(dxs0**2+dys0**2)

def calculate_spring_forces(xs,ys):
    dxs = np.diff(xs)
    dys = np.diff(ys)
    fxs = k*(dxs-dxs0)
    fys = k*(dys-dys0)
    return fxs,fys

calculate_spring_forces(xs=xs0, ys=ys0)
