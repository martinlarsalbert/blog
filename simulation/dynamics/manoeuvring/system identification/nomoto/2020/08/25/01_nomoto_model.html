First order Nomoto model for ship maneuvering dynamics

Implementation of the so called first order Nomoto model for ship maneuvering dynamics

The first order Nomoto model can be written as: $$ K \cdot \delta=r+T_{1} \cdot \dot{r}+T_{2} \cdot \ddot{r} $$ where $r$ is yaw rate with its time derivatives and $\delta$ is the rudder angle. $K$, $T_{1}$ and $T_{1}$ are the coefficients describing the hydrodynamics of the ship.

For slow manoeuvres this equation can be further simplified by removing the $\ddot{r}$ term: $$ K \cdot \delta=r+T_{1} \cdot \dot{r}$$

Simulate a pull out manoeuvre 

#collapse
K = 0.1
delta = 0.15
T_1 = 10

def calculate_r1d(K,T_1,delta,r):
    return (K*delta - r)/T_1 
    
def nomoto(states,t):
    
    # states:
    # [psi, r, r1d]
    # K*delta = r + T_1*r1d
    # --> r = K*delta - T_1*r1d + T_2*r2d
    # --> r1d = (K*delta - r - T_2*r2d)/T_1 
    
    psi = states[0]
    r = states[1]
    
    r1d_ = calculate_r1d(K=K, T_1=T_1, delta=delta, r=r)
    
    d_states_dt = np.array([r,r1d_])
    
    return d_states_dt

states_0 = np.array([0,0])
t = np.linspace(0,30,100)
states = odeint(func = nomoto,y0 = states_0,t = t)
psi = states[:,0]
r = states[:,1]

fig,ax=plt.subplots()
ax.plot(t,np.rad2deg(psi));
ax.grid(True)
ax.set_title('Simulation with Nomoto model $\delta$=%0.1f degrees' %np.rad2deg(delta))
ax.set_ylabel('$\psi$ [deg]')
ax.set_xlabel('time [s]');

Simulate a ZigZag test 

#collapse

K = 0.2
delta = 0.15
T_1 = 10

def nomoto_zigzag(states,t, K, T_1, angle=np.deg2rad(10)):
    
    # states:
    # [psi, r, r1d]
    # K*delta = r + T_1*r1d
    # --> r = K*delta - T_1*r1d + T_2*r2d
    # --> r1d = (K*delta - r - T_2*r2d)/T_1 
    
    psi = states[0]
    r = states[1]
    
    #ZigZag autopilot:
    sign=-1
    psi_pluss = ((psi>=0) and (psi<angle))
    psi_minus = ((psi>=-angle) and (psi<=0))
    
    if (
        (psi_pluss and (r>=0)) |
        (psi<-angle) |
        (psi_minus and (r>=0))
        
        
    ):
        sign=1   
    delta_=sign*delta
    
    #Nomoto equation:
    r1d_ = (K*delta_ - r)/T_1 
    d_states_dt = np.array([r,r1d_])
    
    return d_states_dt

angle=np.deg2rad(10)
states_0 = np.array([0,0])
t = np.linspace(0,100,100)
states = odeint(func = nomoto_zigzag,y0 = states_0,t = t, args=(K,T_1,angle))
psi = states[:,0]
r = states[:,1]

fig,ax=plt.subplots()
ax.plot(t,np.rad2deg(psi));
ax.grid(True)
ax.set_title('Simulation ZigZag%0.0f/%0.0f with Nomoto model' % (np.rad2deg(angle),np.rad2deg(angle)))
ax.set_ylabel('$\psi$ [deg]')
ax.set_xlabel('time [s]');

def nomoto_replay(states, t, K, T_1, deltas):
    
    # states:
    # [psi, r, r1d]
    # K*delta = r + T_1*r1d
    # --> r = K*delta - T_1*r1d + T_2*r2d
    # --> r1d = (K*delta - r - T_2*r2d)/T_1 
    
    psi = states[0]
    r = states[1]
    
    i=np.argmin(np.abs(deltas.index-t))
    delta=deltas.iloc[i]
    
    r1d_ = calculate_r1d(K=K, T_1=T_1, delta=delta, r=r)
    
    d_states_dt = np.array([r,r1d_])
    
    return d_states_dt

def calculate_r1d_T2(K,T_1,T_2,delta,r,r1d):
    # r = K*delta - T_1*r1d + T_2*r2d
    #--> r2d = (r-K*delta + T_1*r1d)/T_2
    
    return (r-K*delta + T_1*r1d)/T_2
